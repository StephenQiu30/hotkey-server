# PLAN-010 Schema 升级

## 范围与停止条件

本手册只新增 `event_contents` 的当前 Content 归属唯一索引、`event_clustering_decisions` append-only 决策表和 `event_governance_audits` append-only 审计表。新环境使用当前完整 `db/schema.sql` 初始化，不执行本手册；既有库升级必须先停止 API/worker，并使用可丢弃 `.env` 验收配置。任何 backup、preflight、DDL、`db verify` 或 restore 失败都停止并回退，不手工拼接部分修复。

## 固定历史基线、备份与只读 preflight

固定 PLAN-009 历史基线为 `7cb8148`。演练必须创建 detached worktree，备份为受保护路径，绝不提交 dump、CSV、DSN 或凭据：

```bash
export PLAN009_BASELINE=7cb8148
export PLAN009_WORKTREE="$(mktemp -d /tmp/hotkey-plan010-plan009.XXXXXX)"
export PLAN010_TARGET=4d5fba5
export PLAN010_WORKTREE="$(mktemp -d /tmp/hotkey-plan010-target.XXXXXX)"
git worktree add --detach "$PLAN009_WORKTREE" "$PLAN009_BASELINE"
git worktree add --detach "$PLAN010_WORKTREE" "$PLAN010_TARGET"
trap 'git worktree remove --force "$PLAN010_WORKTREE"; git worktree remove --force "$PLAN009_WORKTREE"' EXIT

pg_dump "$HOTKEY_DATABASE_URL" --format=custom --file=/secure-backups/hotkey-before-plan010.dump
pg_restore --list /secure-backups/hotkey-before-plan010.dump
HOTKEY_DATABASE_URL="$HOTKEY_DATABASE_URL" go -C "$PLAN009_WORKTREE" run ./cmd/hotkey db verify

psql "$HOTKEY_DATABASE_URL" -v ON_ERROR_STOP=1 -c "
SELECT
  to_regclass('public.event_clustering_decisions') IS NOT NULL AS decisions_exists,
  to_regclass('public.event_governance_audits') IS NOT NULL AS audits_exists,
  to_regclass('public.event_contents_one_current_event_uq') IS NOT NULL AS current_member_index_exists;"

psql "$HOTKEY_DATABASE_URL" -v ON_ERROR_STOP=1 -c "
  SELECT content_id, count(*) AS current_members
  FROM event_contents
  WHERE evidence_role <> 'duplicate'
  GROUP BY content_id
  HAVING count(*) > 1;"
```

若历史库已经包含任一 PLAN-010 对象，停止并转为一次性核对，不重复执行；若 `event_contents` 存在同一 Content 的多个非 duplicate 成员，必须先由事件负责人按最高 membership_score/人工锁规则形成书面清单，不能由 SQL 静默删除。

## 受控升级

升级必须在一个 `psql -v ON_ERROR_STOP=1` 事务中执行。下面的 block 与固定 `PLAN010_TARGET` 的 Schema/catalog 完全一致，顺序为：创建唯一索引、创建决策表、创建幂等索引和 Content/created_at 索引、创建治理审计表及索引，随后用该 release 的 `db verify` 验证。不得把后续 PLAN 的 Schema 一并加入，也不运行时 DDL，不改变旧事件、成员或审计事实。

```bash
psql "$HOTKEY_DATABASE_URL" -v ON_ERROR_STOP=1 <<'SQL'
BEGIN;

CREATE UNIQUE INDEX event_contents_one_current_event_uq
  ON event_contents(content_id) WHERE evidence_role <> 'duplicate';

CREATE TABLE event_clustering_decisions (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  content_id bigint NOT NULL REFERENCES contents(id) ON DELETE RESTRICT,
  candidate_event_id bigint REFERENCES events(id) ON DELETE SET NULL,
  candidate_event_key varchar(96) NOT NULL,
  clustering_version varchar(64) NOT NULL,
  feature_input_hash char(64) NOT NULL,
  channel varchar(16) NOT NULL CHECK (channel IN ('lexical','temporal','fingerprint','vector')),
  candidate_rank integer NOT NULL CHECK (candidate_rank >= 0 AND candidate_rank <= 50),
  entity_action_score numeric(5,2) NOT NULL CHECK (entity_action_score BETWEEN 0 AND 100),
  semantic_score numeric(5,2) NOT NULL CHECK (semantic_score BETWEEN 0 AND 100),
  temporal_score numeric(5,2) NOT NULL CHECK (temporal_score BETWEEN 0 AND 100),
  location_score numeric(5,2) NOT NULL CHECK (location_score BETWEEN 0 AND 100),
  source_context_score numeric(5,2) NOT NULL CHECK (source_context_score BETWEEN 0 AND 100),
  membership_score numeric(5,2) NOT NULL CHECK (membership_score BETWEEN 0 AND 100),
  decision varchar(16) NOT NULL CHECK (decision IN ('accepted','review','rejected','new_event')),
  decision_origin varchar(16) NOT NULL CHECK (decision_origin IN ('rule','model','user')),
  reason_codes text[] NOT NULL DEFAULT '{}',
  feature_snapshot jsonb NOT NULL DEFAULT '{}'::jsonb CHECK (jsonb_typeof(feature_snapshot) = 'object'),
  evidence_content_ids bigint[] NOT NULL DEFAULT '{}',
  actor_user_id bigint REFERENCES users(id) ON DELETE SET NULL,
  created_at timestamptz NOT NULL DEFAULT now(),
  CHECK ((decision = 'new_event' AND candidate_event_id IS NULL AND candidate_event_key = '__new_event__') OR (decision <> 'new_event' AND candidate_event_id IS NOT NULL AND candidate_event_key <> '__new_event__'))
);
CREATE UNIQUE INDEX event_clustering_decisions_idempotency_uq
  ON event_clustering_decisions(content_id, clustering_version, feature_input_hash, candidate_event_key);
CREATE INDEX event_clustering_decisions_content_idx
  ON event_clustering_decisions(content_id, created_at DESC, id DESC);

CREATE TABLE event_governance_audits (
  id bigint GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  event_id bigint NOT NULL REFERENCES events(id) ON DELETE RESTRICT,
  action varchar(40) NOT NULL CHECK (action IN ('lifecycle_transition','merge','split','member_move','manual_lock','manual_unlock','evidence_recompute','downstream_reconcile_required','deduplicated')),
  actor_user_id bigint REFERENCES users(id) ON DELETE SET NULL,
  reason_code varchar(64) NOT NULL,
  from_status varchar(16) CHECK (from_status IS NULL OR from_status IN ('detected','active','cooling','closed','archived','rejected','merged')),
  to_status varchar(16) CHECK (to_status IS NULL OR to_status IN ('detected','active','cooling','closed','archived','rejected','merged')),
  source_event_id bigint REFERENCES events(id) ON DELETE SET NULL,
  target_event_id bigint REFERENCES events(id) ON DELETE SET NULL,
  expected_version bigint,
  metadata jsonb NOT NULL DEFAULT '{}'::jsonb CHECK (jsonb_typeof(metadata) = 'object'),
  created_at timestamptz NOT NULL DEFAULT now()
);
CREATE INDEX event_governance_audits_event_idx
  ON event_governance_audits(event_id, created_at DESC, id DESC);

COMMIT;
SQL

HOTKEY_DATABASE_URL="$HOTKEY_DATABASE_URL" go -C "$PLAN010_WORKTREE" run ./cmd/hotkey db verify
```

## 回退

若升级未提交，直接回滚事务。若已提交，先停止服务并确认 custom dump 为 PLAN-009 基线。先在可丢弃副本上证明未准备 restore 会原子失败；随后只移除 PLAN-010 新对象，再恢复备份。不得删除或重建 PLAN-009 的 `events`、`event_contents`、`monitor_events` 数据。

```bash
if pg_restore --single-transaction --clean --if-exists --no-owner \
  --dbname="$HOTKEY_DATABASE_URL" /secure-backups/hotkey-before-plan010.dump; then
  echo "unexpected PLAN-010 restore success without preparation" >&2
  exit 1
fi

psql "$HOTKEY_DATABASE_URL" -v ON_ERROR_STOP=1 <<'SQL'
BEGIN;

DROP TABLE IF EXISTS event_clustering_decisions;
DROP TABLE IF EXISTS event_governance_audits;
DROP INDEX IF EXISTS event_contents_one_current_event_uq;

COMMIT;
SQL

pg_restore --single-transaction --clean --if-exists --no-owner \
  --dbname="$HOTKEY_DATABASE_URL" /secure-backups/hotkey-before-plan010.dump

# pg_restore 等价改写部分谓词；重建 PLAN-009 的两个 canonical 索引后再校验历史 release。
psql "$HOTKEY_DATABASE_URL" -v ON_ERROR_STOP=1 <<'SQL'
BEGIN;

DROP INDEX IF EXISTS ai_runs_reuse_inflight_uq;
CREATE UNIQUE INDEX ai_runs_reuse_inflight_uq
  ON ai_runs(reuse_key)
  WHERE status IN ('queued','running','validating','retry_wait');

DROP INDEX IF EXISTS monitor_rules_relevance_approved_lexical_idx;
CREATE INDEX monitor_rules_relevance_approved_lexical_idx
  ON monitor_rules(lower(value), config_version_id, origin, weight DESC, id)
  WHERE enabled AND approval_status = 'approved'
    AND rule_type IN ('keyword','phrase','entity','exclude_keyword');

COMMIT;
SQL

HOTKEY_DATABASE_URL="$HOTKEY_DATABASE_URL" go -C "$PLAN009_WORKTREE" run ./cmd/hotkey db verify
```
